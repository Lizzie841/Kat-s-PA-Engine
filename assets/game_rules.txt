#Empty effects won't do anything
action = {
	type = "state_transfer"
	#THIS = nation asker
	#FROM = nation target
	#main = state id
	limit = {
		crisis_exist = no
		THIS = {
			tag != FROM
			number_of_states > 1
			is_independant = yes
			ai = no
			war = no
		}
		FROM = {
			is_independant = yes
			ai = no
			war = no
		}
		any_owned_province = { controlled_by = THIS }
	}
	effect = {
		any_owned_province = { secede_province = FROM }
	}
}
